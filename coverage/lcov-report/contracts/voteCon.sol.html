pragma solidity ^0.8.9;
pragma abicoder v2;
 
contract voteCon 
{  
     struct Voter {
        mapping(uint => bool) voted; 
        mapping(uint => uint) vote;
    }
 
    struct Candidate {
        string name;   
        address candidateAddress;
        uint number;
        uint voteCount; 
    }
 
    struct Voting {
        uint numberOfVating;
        uint candidatesAmount;
        uint votersAmount;
    }
    
    address public chairperson;
    mapping(address => Voter) voters;
    mapping(uint => mapping(uint => Candidate)) public candidates;
    mapping(uint => Voting) public votings;
 
    uint startTime;
    mapping(uint => uint) endTime;
    uint voteCost = 100;
    uint timeOfVoting = 60;//259200;
    uint votingAmount;
 
    constructor() {
        chairperson = msg.sender;
    }
 
    modifier onlySmartContractOwner() {
        require(msg.sender == chairperson, "Only chairperson can start and end the voting");
        _;
    }
 
    modifier votingIsOpen() {
        require(block.timestamp <= endTime[votingAmount], "Voting is closed");
        _;
    }
 
    function createVoting(string[3] memory candidateNames, address[3] memory candidateAddresses) public onlySmartContractOwner {
        votingAmount++;
        votings[votingAmount].candidatesAmount =candidateNames.length;
        votings[votingAmount].numberOfVating = votingAmount;
        for (uint i = 0; i < votings[votingAmount].candidatesAmount; i++) {
            candidates[votingAmount][i+1].name = candidateNames[i];
            candidates[votingAmount][i+1].candidateAddress = candidateAddresses[i];
            candidates[votingAmount][i+1].number = i+1;
        }
         startTime = block.timestamp;
         endTime[votingAmount] = startTime + timeOfVoting;
    }
 
    function vote(uint votingNumber, uint candidateNumber) public payable votingIsOpen {
        Voter storage sender = voters[msg.sender];
        require(!sender.voted[votingNumber], "Already voted.");
        require(msg.value == voteCost, "Invalid Amount");
        sender.voted[votingNumber] = true;
        sender.vote[votingNumber] = candidateNumber;
        candidates[votingNumber][candidateNumber].voteCount++;
    }
 
    function closeVoting(uint votingNumber) public payable onlySmartContractOwner {
        uint winningCandidate = candidates[votingNumber][1].voteCount;
        uint winnerNumber = candidates[votingNumber][1].number;
        for (uint i = 2; i <= votings[votingNumber].candidatesAmount; i++) {
            if (winningCandidate < candidates[votingNumber][i].voteCount) {
                winningCandidate = candidates[votingNumber][i].voteCount;
                winnerNumber = candidates[votingNumber][i].number;
            }
        }
 
        address payable winner = payable(candidates[votingNumber][winnerNumber].candidateAddress);
        winner.transfer(address(this).balance * 9/10);
    }
 
    function withdrawBalance(address payable receiver) onlySmartContractOwner public payable {
        receiver.transfer(address(this).balance);
    }
 
    function votingDates(uint votingNumber) public view returns(uint[] memory votes) {
        uint[] memory _votes;
        for (uint i = 1; i<=votings[votingNumber].candidatesAmount; i++) {
            _votes[i-1] = candidates[votingNumber][i].voteCount;
        }
        return(_votes);
    }
 
    function votingDate(uint votingNumber) public view returns(uint _numberOfVating, uint _candidatesAmount, uint _votersAmount) {
        return(votings[votingNumber].numberOfVating, votings[votingNumber].candidatesAmount, votings[votingNumber].votersAmount);
    }
 
    function balance() public view returns(uint _balance) {
        return(address(this).balance);
    }
}
